from services.execution.place_order import place_order, place_order_async  # centralized order submit

#!/usr/bin/env python3
"""
Automatically replace direct `.create_order` calls outside place_order.py
with `place_order(...)` wrapper calls.
"""

import re
from pathlib import Path

# Directory of your project (adjust if needed)
ROOT = Path(__file__).resolve().parent

# Skip these directories/files
SKIP_DIRS = {"tests", "__pycache__"}
PLACE_ORDER_FILE = "place_order.py"

# Regex to find create_order calls
CREATE_ORDER_RE = re.compile(r"\b(\w+)\.create_order\s*\(")

def process_file(file_path: Path):
    if file_path.name == PLACE_ORDER_FILE:
        return False  # skip place_order.py itself

    changed = False
    text = file_path.read_text(encoding="utf-8")
    new_lines = []

    for line in text.splitlines():
        if ".create_order" in line:
            # Replace with place_order call
            new_line = CREATE_ORDER_RE.sub("place_order(", line)
            changed = True
            new_lines.append(new_line)
        else:
            new_lines.append(line)

    if changed:
        file_path.write_text("\n".join(new_lines), encoding="utf-8")
        print(f"[FIXED] {file_path}")

    return changed

def main():
    py_files = [p for p in ROOT.rglob("*.py") if not any(s in p.parts for s in SKIP_DIRS)]
    total_fixed = 0
    for f in py_files:
        if process_file(f):
            total_fixed += 1

    print(f"\nTotal files fixed: {total_fixed}")
    print("✅ All direct create_order calls replaced with place_order")

if __name__ == "__main__":
    main()
