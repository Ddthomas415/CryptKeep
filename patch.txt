from pathlib import Path
import re

def read(p: str) -> str:
    P = Path(p)
    return P.read_text(encoding="utf-8", errors="replace") if P.exists() else ""

def safe_patch(path: str, marker: str, fn):
    P = Path(path)
    if not P.exists():
        print(f"SKIP missing: {path}")
        return False
    t = read(path)
    if marker in t:
        print(f"SKIP already: {path}")
        return False
    t2 = fn(t)
    if t2 == t:
        print(f"SKIP nochange: {path}")
        return False
    P.write_text(t2, encoding="utf-8")
    print(f"PATCHED: {path}")
    return True

def append_checkpoint(title: str, body: str):
    ck = Path("CHECKPOINTS.md")
    if not ck.exists():
        return
    t = read("CHECKPOINTS.md")
    if title in t:
        return
    ck.write_text(t.rstrip() + "\n\n" + (title + "\n" + body).lstrip("\n"), encoding="utf-8")
